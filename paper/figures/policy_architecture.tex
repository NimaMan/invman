% PAGE 2 -- NN BACKBONE: state vector -> hidden layer -> latent outputs
% (adapted from the prior version's neural-network figure)
% =====================================================================
\begin{tikzpicture}[x=2.7cm,y=1.4cm]
  \readlist\Nnod{3,6,4}        % nodes per layer: d (=L), H, K (latent outputs)

  \readlist\Cstr{\strut x,h,u} % coefficient symbol per layer
  \def\yshift{0.5}

  \foreachitem \N \in \Nnod{
    \def\lay{\Ncnt}
    \pgfmathsetmacro\prev{int(\Ncnt-1)}
    \foreach \i [evaluate={\c=int(\i==\N); \y=\N/2-\i-\c*\yshift;
                 \index=(\i<\N?int(\i):"\Nstr[\lay]");
                 \x=\lay; }] in {1,...,\N}{
      \ifnum\lay=1
        \node[node in]     (N\lay-\i) at (\x,\y) {$\Cstr[\lay]_{\index}$};
      \else\ifnum\lay=\Nnodlen
        \node[node out]    (N\lay-\i) at (\x,\y) {$\Cstr[\lay]_{\index}$};
      \else
        \node[node hidden] (N\lay-\i) at (\x,\y) {$\Cstr[\lay]_{\index}$};
      \fi\fi
      \ifnum\lay>1
        \foreach \j in {1,...,\Nnod[\prev]}{
          \draw[connect,white,line width=1.2] (N\prev-\j) -- (N\lay-\i);
          \draw[connect] (N\prev-\j) -- (N\lay-\i);
        }
      \fi
    }
    \path (N\lay-\N) --++ (0,1+\yshift) node[midway,scale=1.5] {$\vdots$};
  }
  % column captions at a single consistent height just above the tallest column
  \node[align=center,mygreen!60!black] at (1,2.9) {normalized\\[-0.2em]state $\mathbf{x}$};

\end{tikzpicture}


% =====================================================================
% PAGE 3 -- DECODER HEADS: latent outputs u -> integer order a
% Each head is part of the policy; all map to a valid (non-negative integer) action.
% =====================================================================
\begin{tikzpicture}[every node/.style={align=left},
                    head/.style={anchor=north west,mydarkblue,align=left,
                                 rounded corners=2pt,draw=mydarkblue!25,fill=mydarkblue!4,
                                 inner sep=4pt,text width=82mm}]
  % Four heads stacked with a uniform edge gap; relative positioning absorbs their
  % different heights, so they neither collide nor leave an uneven gap.
  \node[head] (d0) at (0,0)
    {\textbf{categorical} \;(scores $y_0,\dots,y_{\bar q}$):\\[2pt]
     $a=\argmax_{q\in\{0,\dots,\bar q\}} y_q$};
  \node[head,below=2.4mm of d0.south west,anchor=north west] (d1)
    {\textbf{soft-gated direct} \;(gate $g$, quantity $q$):\\[2pt]
     $a=\big\lfloor\, \sigma(g)\cdot\mathrm{softplus}(q)\,\big\rceil_{\ge0}$};
  \node[head,below=2.4mm of d1.south west,anchor=north west] (d2)
    {\textbf{soft-gated ordinal} \;(gate $g$, thresholds $o_1,\dots,o_M$):\\[2pt]
     $a=\Big\lfloor\, \sigma(g)\sum_{k=1}^{M}\sigma(o_k)\,\Big\rceil_{\ge0}$};
  \node[head,below=2.4mm of d2.south west,anchor=north west] (d3)
    {\textbf{soft tree} \;(leaf prob.\ $\pi_\ell$, leaf quantity $q_\ell$):\\[2pt]
     $a=\Big\lfloor\, \sum_{\ell}\pi_\ell(\mathbf{x})\,q_\ell(\mathbf{x})\,\Big\rceil_{\ge0}$};

  % latent-outputs box (left) and valid-action box (right), vertically centred on the stack
  \node[fit=(d0)(d3),inner sep=0pt] (stack) {};
  \node[stage mid,left=10mm of stack.west] (u) {latent\\outputs\\$\mathbf{u}$};
  \node[stage out,right=10mm of stack.east] (a) {\textbf{valid}\\\textbf{action} $a_t$};
  \foreach \d in {d0,d1,d2,d3}{ \draw[flow] (u.east) -- (\d.west); }

  \node[mydarkblue,below=5mm of d3.south west,anchor=north west,align=left]
    {\footnotesize $\sigma$: logistic; \;$\mathrm{softplus}(z)=\ln(1+e^z)$; \;$\lfloor\cdot\rceil_{\ge0}$: round to nearest non-negative integer.};
\end{tikzpicture}


% =====================================================================
% PAGE 4 -- CMA-ES TRAINING LOOP (generational optimization cycle)
% =====================================================================
\begin{tikzpicture}[x=1cm,y=1cm]
  \node[stage mid] (dist) at (0,0)    {search distribution\\$\mathcal{N}(m_g,\sigma_g^2 C_g)$};
  \node[stage in]  (samp) at (4.1,0)  {sample population\\$\theta_1,\dots,\theta_\lambda$ \,($\lambda=64$)};
  \node[stage dec] (eval) at (8.4,0)  {evaluate\\\footnotesize each $\theta_i\!\to\!\pi_{\theta_i}\!\to$ short rollouts\\\footnotesize $\to\ \hat C_T(\theta_i)$ (cost)};
  \node[stage mid] (rank) at (12.5,0) {rank \& select\\\footnotesize keep best $\mu$ by cost};
  \node[stage out] (upd)  at (16.1,0) {update\\$m_{g+1},\,\sigma,\,C$};
  \draw[flow] (dist) -- (samp);
  \draw[flow] (samp) -- (eval);
  \draw[flow] (eval) -- (rank);
  \draw[flow] (rank) -- (upd);
  % return arrow: next generation
  \draw[flow] (upd.south) -- ++(0,-1.2) -| (dist.south);
  \node[mydarkblue] at (8.05,-1.55) {\footnotesize next generation $g\!\to\!g+1$};
  % annotations
  \node[myorange!60!black,above=2pt of eval.north] {\footnotesize only stage that touches the environment};
  \node[mydarkblue,below=15mm of eval.south,anchor=north,align=center]
    {\footnotesize Gradient-free: the update uses only the \emph{ranking} of the costs $\hat C_T(\theta_i)$, not their gradients.};
\end{tikzpicture}


% =====================================================================
% PAGE 5 -- SOFT TREE (oblique soft decision tree with linear leaves)
% =====================================================================
\begin{tikzpicture}[x=1cm,y=1cm]
  \node[stage in] (x) at (0,2.0) {normalized state $\mathbf{x}=\mathbf{S}_t/\kappa$};
  % internal (split) nodes
  \node[node hidden] (r)  at (0,0)    {$n_0$};
  \node[node hidden] (n1) at (-2.6,-2){$n_1$};
  \node[node hidden] (n2) at (2.6,-2) {$n_2$};
  % leaves (linear-leaf quantities)
  \node[stage mid] (l1) at (-3.9,-4) {$q_1(\mathbf{x})$};
  \node[stage mid] (l2) at (-1.3,-4) {$q_2(\mathbf{x})$};
  \node[stage mid] (l3) at (1.3,-4)  {$q_3(\mathbf{x})$};
  \node[stage mid] (l4) at (3.9,-4)  {$q_4(\mathbf{x})$};
  % input to root
  \draw[flow] (x) -- (r);
  % oblique-split routing edges with probabilities
  \draw[connect arrow] (r)  -- (n1) node[midway,above left,font=\footnotesize]  {$1-\sigma(\cdot)$};
  \draw[connect arrow] (r)  -- (n2) node[midway,above right,font=\footnotesize] {$\sigma(w_0^\top\mathbf{x}+b_0)$};
  \draw[connect arrow] (n1) -- (l1) node[midway,left,font=\footnotesize]  {$1-\sigma(\cdot)$};
  \draw[connect arrow] (n1) -- (l2) node[midway,right,font=\footnotesize] {$\sigma(\cdot)$};
  \draw[connect arrow] (n2) -- (l3);
  \draw[connect arrow] (n2) -- (l4);
  % output: path-probability mixture, rounded
  \node[stage out] (out) at (0,-6.3) {$a=\big\lfloor\, \sum_{\ell}\pi_\ell(\mathbf{x})\,q_\ell(\mathbf{x})\,\big\rceil_{\ge0}$};
  % the four leaves jointly determine the order: group them, single connector to the mixture
  \begin{scope}[on background layer]
    \node[draw=mydarkblue!55,dashed,rounded corners=5pt,inner sep=8pt,
          fit=(l1)(l2)(l3)(l4)] (leafgrp) {};
  \end{scope}
  \draw[flow] (leafgrp.south) -- (out.north)
        node[midway,right=2pt,font=\footnotesize,mydarkblue] {combined order};
  \node[mydarkblue,below=3pt of out.south,align=center]
    {\footnotesize Each leaf $\ell$ has path probability $\pi_\ell(\mathbf{x})=\prod(\text{routing probs on its root-to-leaf path})$, with $\sum_\ell\pi_\ell=1$;\\
     the action is the soft mixture of the leaf quantities, rounded to a valid order.};
\end{tikzpicture}


% =====================================================================
% ENVIRONMENT FLOW-NETWORK DIAGRAMS (one per paper problem).
% Visual language reused from the policy figures above:
%   - "stage in"   (green)  = external supply source / manufacturer
%   - "stage mid"  (blue)   = on-hand stock point / warehouse / retailer
%   - "stage dec"  (orange) = demand sink
% Each interaction is EVENT-BASED, drawn as two distinct arrows:
%   - "reqflow"  (thin dashed green, information/control) = the stock point SENDS
%       an order *request* q_t to the supplier (request->supplier direction).
%   - "matflow"  (solid blue, goods)                      = the supplier SHIPS;
%       supplies ARRIVE after the lead time (supplier->stock direction).
% "demandflow" (orange) = demand served from stock; "lostflow"/"boflow" (red,
% dashed) = unmet demand lost or backordered.  Each figure is wrapped in a dashed
% rounded "envbox" naming the environment, and a one-line numbered event sequence
% is printed beneath the box (observe -> request -> arrive -> demand -> lose/
% backorder -> cost), matching the flownet timing layer.
% Costs are written in colour next to where they are charged:
%   holding h (blue), lost-sales p / backorder b (red), order/fixed cost (dark).
% =====================================================================
% Node text and edge labels are deliberately set at \small (body-size after the
% ~1x resizebox in the manuscript), with generously sized nodes so the text never
% overflows. Edge labels carry a small white background fill so that, even if an
% arc passes near them, the text reads cleanly (no text-on-arc collision).

  envsupply/.style={stage in,font=\small,minimum height=15mm,minimum width=30mm},
